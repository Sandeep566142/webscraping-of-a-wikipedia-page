def scrape_wikipedia(url):
    # Send a GET request to the Wikipedia page
    response = requests.get(url)
    
    # Parse the HTML content using BeautifulSoup
    soup = BeautifulSoup(response.text, 'html.parser')
    
    # Extract headings and text content
    headings = [heading.text for heading in soup.find_all(['h1', 'h2', 'h3', 'h4', 'h5', 'h6'])]
    text_content = [paragraph.text for paragraph in soup.find_all('p')]
    
    return headings, text_content

def summarize_text_with_llama(input_text):
    # LLama2 model
    llm=HuggingFaceHub(repo_id="Falconsai/text_summarization",model_kwargs={"temperature":0.1,"max_length":200})

    # Prompt Template
    template = """
        Generate a concise summary for the text:{input_text}
        """
    
    prompt = PromptTemplate(input_variables=["input_text"],
                            template=template)

    # Generate the response from the LLama 2 model
    response = llm(prompt.format(input_text=input_text))

    return response
